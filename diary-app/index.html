const PROXY_URL = 'https://corsproxy.io/?'; const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbyGl53JqWXfLi2gacp8Cw78nE9MePMSyvr9fe385YCrzsybjHpC5I1CqcvWunAKiko8/exec'; function saveEntry() { const title = encodeURIComponent(document.getElementById('title').value.trim()); const text = encodeURIComponent(document.getElementById('content').value.trim()); if (!title || !text) { alert("Please enter title and text!"); return; } const fullURL = `${PROXY_URL}${SCRIPT_URL}?title=${title}&text=${text}`; fetch(fullURL) .then(res => res.json()) .then(response => { document.getElementById('status').innerText = "✅ " + response.message; document.getElementById('title').value = ''; document.getElementById('content').value = ''; }) .catch(err => { document.getElementById('status').innerText = "❌ Error saving entry: " + err.message; console.error(err); }); }
